const SHEET_ID = "YOUR_SHEET_ID_HERE"; const SHEETS = { income:"Income", expense:"Expenses", invest:"Investments" }; const HEADERS = { income: ["Date","Source","Description","Amount","Month","Year","Notes"], expense: ["Date","Category","Description","Amount","Month","Year"], invest: ["Date","Type","Name","Amount","Units","Month","Year","Notes"] }; function doGet(e) { const action = (e.parameter||{}).action||""; const ss = SpreadsheetApp.openById(SHEET_ID); ensureSheets(ss); if (action==="ping") return out({ok:true,message:"Connected!"}); if (action==="fetch") { const result={}; Object.keys(SHEETS).forEach(key=>{ const sh=ss.getSheetByName(SHEETS[key]); const vals=sh.getDataRange().getValues(); if(vals.length<=1){result[key]=[];return;} const h=vals[0]; result[key]=vals.slice(1).map(r=>{let o={};h.forEach((k,i)=>o[k]=r[i]);return o;}); }); return out({ok:true,data:result}); } return out({ok:false,error:"Unknown GET action"}); } function doPost(e) { try { const ss=SpreadsheetApp.openById(SHEET_ID); ensureSheets(ss); const d=JSON.parse(e.postData.contents); if(d.action==="add"){ const sh=ss.getSheetByName(SHEETS[d.type]); if(!sh) return out({ok:false,error:"Unknown type: "+d.type}); const dt=new Date(d.date); const m=dt.getMonth()+1; const y=dt.getFullYear(); if(d.type==="income") sh.appendRow([d.date,d.source||"",d.description||"",Number(d.amount),m,y,d.notes||""]); if(d.type==="expense") sh.appendRow([d.date,d.category||"",d.description||"",Number(d.amount),m,y]); if(d.type==="invest") sh.appendRow([d.date,d.itype||"",d.name||"",Number(d.amount),d.units||"",m,y,d.notes||""]); return out({ok:true}); } return out({ok:false,error:"Unknown POST action: "+d.action}); } catch(err){return out({ok:false,error:err.message});} } function ensureSheets(ss){ Object.keys(SHEETS).forEach(key=>{ let sh=ss.getSheetByName(SHEETS[key]); if(!sh){sh=ss.insertSheet(SHEETS[key]);sh.appendRow(HEADERS[key]);sh.setFrozenRows(1);} }); } function out(d){return ContentService.createTextOutput(JSON.stringify(d)).setMimeType(ContentService.MimeType.JSON);}
